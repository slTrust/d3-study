序数比例尺

// 离散的映射关系
var index = [0, 1, 2, 3, 4];
var color = ["red", "blue", "green", "yellow", "black"];

// red对应0， blue对应 1
var ordinal = d3.scale.ordinal()
    .domain(index)
    .range(color);

ordinal(0); //返回 red
ordinal(2); //返回 green
ordinal(4); //返回 black
